import random
from textblob import TextBlob
from googletrans import Translator

#Lista de saludos y respuestas
saludos=["hola","que tal","buenos dias","buenas tardes", "buen dia"]
respuesta_saludos=["hola, como estas?",
                   "como puedo ayudarte",
                   "hola, como te va",
                   "que hay?"]

despedida=["adios","bye","gracias","nos vemos","hasta luego","salir"]
respuesta_despedida=[
    "adios","hasta pronto",
    "fue un placer ayudarte",
    "cuidate",
    "nos vemos luego"
    "bye"
]

pregunta1=["¿como estas?"]
pregunta2=[ "¿que haces?"]
pregunta3=["¿puedes contarme un chiste?"]
pregunta4=["¿te gusta hablar conmigo?"]
respuestas_preguntas1=["¡Estoy genial! Gracias por preguntar",
                       "¡Me encuentro muy bien!",
                       "¡Estoy bien!"]
respuestas_preguntas2=["¡Estoy aqui,listo para ayudarte!",
                       "¡Esperando para ayudarte en lo que necesites",
                       "¡Listo para empezar a ayudarte!"]
respuestas_preguntas3=["¡Claro!, ¿por que la gallina cruzo la calle? para llegar al otro lado",
                       "¿Por que la ballena no se subio a la combi? porque va llena"]
respuestas_preguntas4=["¡Por supuesto! eres mi amigo de confianza",
                       "¡obvio microbio!",
                       "¡Me encanta conversar contigo"]

preguntas_ia=["¿como te llamas?"]
respuesta_ia=["Soy chato, tu IA de confianza",
              "Me llamo Chato, soy tu chatbot favorito"]

#mensaje de bienvenida del chatbot
print("Hola soy Chato, tu Chatbot de confianza, si deseas"
      "terminar escribe 'salir")

while True:
    usuario=input("Tu:").lower().strip()

    if usuario in saludos:
        print("chatbot:",random.choice(respuesta_saludos),"\n ¿como estas hoy?" )
        translator=Translator()
        text=input(("Tu:"))
        traduccion= translator.translate(text,dest='en').text

        analizar= TextBlob(traduccion)
        sentimiento= analizar.sentiment
        if sentimiento.polarity == 0:
            print("No encuentro un sentimiento en tu oracion")
        elif sentimiento.polarity <= -0.5 and sentimiento.polarity < 0:
            print("Lamento que te sientas asi, muchas veces los sentimientos malos se apoderan de nosotros, ¡animo!"
                  ", solo es un mal momento")
        elif sentimiento.polarity < -0.5:
            print("¿te encuentras bien?, si necesitas ayuda intenta hablar con alguien de confianza.")
        elif sentimiento.polarity > 0 and sentimiento.polarity <= 0.5:
            print("¡Eso es fantastico!")
        elif sentimiento.polarity > 0.5:
            print("¡Que felicidad que te encuentres bien!")





    elif usuario in preguntas_ia:
        print("chatbot:", random.choice(respuesta_ia))
    elif usuario in pregunta1:
        print("chatbot:", random.choice(respuestas_preguntas1))
    elif usuario in pregunta2:
        print("chatbot:", random.choice(respuestas_preguntas2))
    elif usuario in pregunta3:
        print("chatbot:", random.choice(respuestas_preguntas3))
    elif usuario in pregunta4:
        print("chatbot:", random.choice(respuestas_preguntas4))
    elif usuario in despedida:
        print("chatbot:", random.choice(respuesta_despedida))
        break

Glosario de Funciones

random.choice(lista)
Selecciona de forma aleatoria un elemento dentro de una lista.
TextBlob(texto)
Crea un objeto para analizar el texto (permite detectar sentimiento, traducir, corregir, etc.).
Translator()
Crea un traductor de Google que permite traducir textos entre idiomas.
translator.translate(text, dest='en')
Traduce el texto dado (text) al idioma destino indicado (en = inglés).
analizar.sentiment
Obtiene el análisis de sentimiento (polaridad y subjetividad) del texto traducido.
input("mensaje")
Permite recibir texto escrito por el usuario en la consola.
str.lower()
Convierte el texto a minúsculas.
str.strip()
Elimina los espacios en blanco al inicio y al final del texto.
