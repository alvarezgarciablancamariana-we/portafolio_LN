def buscar_respuestas(mensaje):
    palabras = set(word_tokenize(mensaje.lower()))
    coincidencia = palabras & set(respuestas.keys())

    if coincidencia:
        palabra_encontrada = coincidencia.pop()
        return random.choice(respuestas[palabra_encontrada])

    return "No entendi tu mensaje, escribe de nuevo."

print("Chatbot: Hola! escribe 'bye' para salir")

while True:
    mensaje = input("Tu: ")
    if mensaje.lower() == 'bye':
        print("Chatbot: adios")
        break

    respuesta = buscar_respuestas(mensaje)
    print("Chatbot:", respuesta)
