Código del Ejemplo

from deep_translator import GoogleTranslator
import keyboard

def traducir(texto, idioma_destino):
    """Traduce un texto del español al idioma elegido"""
    traduccion = GoogleTranslator(source='es', target=idioma_destino).translate(texto)
    return traduccion

def main():
    print("===  Traductor español/idioma escogido ===")
    print("Presiona:")
    print("  I= Traducir al inglés")
    print("  A= Traducir al alemán")
    print("  F= Traducir al francés")
    print("  Y= Traducir al italiano")
    print("  P= Traducir al portugues")
    print("  x= Salir del programa")

    while True:
        texto = input("\nEscribe un texto en español: ")

        print("\nAhora presiona una tecla para seleccionar el idioma:")
        print("I (inglés) | A (alemán) | F (francés) |Y (italiano) |P portugues) | X (salir del traductor)")

        while True:
            if keyboard.is_pressed('I'):
                print("\n La traducció al inglés:")
                print(traducir(texto, 'en'))
                break
            elif keyboard.is_pressed('A'):
                print("\n La traducció al alemán:")
                print(traducir(texto, 'de'))
                break
            elif keyboard.is_pressed('F'):
                print("\n La traducció al francés:")
                print(traducir(texto, 'fr'))
                break
            elif keyboard.is_pressed('Y'):
                print("\n La traducció al italiano:")
                print(traducir(texto, 'it'))
                break
            elif keyboard.is_pressed('P'):
                print("\n La traducció al portugues:")
                print(traducir(texto, 'pt'))
                break
            elif keyboard.is_pressed('X'):
                print("\n ¡¡¡¡¡¡Saliendo del traductor con exito!!!!!!...")
                return

if __name__ == "__main__":
    main()

Glosario de Funciones

GoogleTranslator
Sirve para traducir textos entre idiomas usando Google Translate.
keyboard
Sirve para detectar qué tecla presiona el usuario.
traducir(texto, idioma_destino)
Sirve para traducir el texto al idioma que elija el usuario.
main()
Sirve para ejecutar el programa, mostrar el menú y controlar el flujo de traducción.
keyboard.is_pressed('I')
Sirve para saber qué idioma eligió el usuario presionando una tecla.
